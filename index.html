function copyResult() { if (lastDraw.length === 0) { alert("請先抽牌"); return; } const q = document.getElementById("question").value.trim(); let text = ""; if (q) text += `我的問題是：${q}\n`; text += `我抽到的雷諾曼牌是：${lastDraw.join("、")}。\n請幫我解讀這組牌的意思。`; // ✅ 方法1：現代 clipboard API if (navigator.clipboard && window.isSecureContext) { navigator.clipboard.writeText(text) .then(() => alert("已複製")) .catch(() => fallbackCopy(text)); } else { // ❗ fallback fallbackCopy(text); } } // ✅ 備用複製（手機最穩） function fallbackCopy(text) { const textarea = document.createElement("textarea"); textarea.value = text; textarea.style.position = "fixed"; textarea.style.left = "-9999px"; document.body.appendChild(textarea); textarea.focus(); textarea.select(); try { document.execCommand("copy"); alert("已複製"); } catch (err) { alert("複製失敗，請手動複製"); } document.body.removeChild(textarea); }
